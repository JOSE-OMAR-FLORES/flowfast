namespace App\Models;

class League extends BaseModel
{

        'name', 'slug', 'sport_id', 'admin_id',
        'registration_fee', 'match_fee', 'status'
    ];

        'is_public' => 'boolean',

    {
        return $this->hasMany(Season::class);
